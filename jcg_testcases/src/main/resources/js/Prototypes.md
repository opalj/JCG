## P2
[//]: # (MAIN: global)
Test basic prototype inheritance.

```json
{
  "directLinks": [
    ["<global>", "P2.foo"]
  ],
  "indirectLinks": []
}
```
```js
// prototypes/P2.js
function foo() {
    return 'animal sound';
}
function Animal() {}
Animal.prototype.speak = foo;

function Dog() {}
Dog.prototype = Object.create(Animal.prototype);
Dog.prototype.constructor = Dog;

const dog = new Dog();
```

## P5
[//]: # (MAIN: global)
Set prototype using Object.setPrototypeOf.

```json
{
  "directLinks": [
    ["<global>", "P5.foo"]
  ],
  "indirectLinks": []
}
```
```js
// prototypes/P5.js

function foo() {
    return 'foo';
}

function Base() {}
Base.prototype = {
    foo: foo
};

function Derived() {}
Object.setPrototypeOf(Derived.prototype, Base.prototype);

const d = new Derived();
d.foo();

```
